[TO RUN THE DEV SERVERS]
SNIPPETS:
run backend:

cd backend
.\venv\Scripts\activate
python main.py 

run frontend:
by building (production build)

Build:
cd frontend
flutter build web --release

Run:
cd frontend/build/web
python -m http.server 8888

Direct run: (debug)
flutter run -d edge
